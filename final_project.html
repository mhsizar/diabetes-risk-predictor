anova<-project2%>%
  select(bmi_class,systolicBP)

Question: Does BMI (Underweight/Normal weight/Overweight) has any effect on Systolic Blood pressure?

Null Hypothesis, H0 : The mean systolic blood pressure (MMHg) is the same in each type of BMI (1=Underweight, 2=Normal weight, 3=overweight).

Alternative Hypothesis, HA : At least people with one category of BMI has not the same mean systolic blood pressure (MMHg) as the others.

#ANOVA Test

bp<- aov(systolicBP~bmi_class,data = anova)
